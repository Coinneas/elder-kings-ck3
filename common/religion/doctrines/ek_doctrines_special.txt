is_nedic_nordic_faith = {
	group = "special"
	
	is_available_on_create = {
		has_doctrine = special_doctrine_is_nedic_nordic_faith
	}

	special_doctrine_is_nedic_nordic_faith = {
		parameters = {
			# Since doctrines cannot modify other faith's views of us, only our view of others, all Nedic-Nordic faiths are given this special doctrine which makes them friendlier to Faiths with the Nedic-Nordic Syncretism tenet.
			hostility_override_tenet_nedic_nordic_syncretism = 1
			nedic_nordic_syncretic_recipient_opinion_active = yes
			opinion_of_nedic_nordic_syncretic_actor_opinion_active = 15
		}
	}
}
